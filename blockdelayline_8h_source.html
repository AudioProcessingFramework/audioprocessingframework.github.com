116 template<typename T, typename Container>
117 BlockDelayLine<T, Container>::BlockDelayLine(size_type block_size
118  , size_type max_delay)
119  : _block_size(block_size)
120  , _max_delay(max_delay)
121  // Minimum number of blocks is 2, even if _max_delay is 0.
122  // With only one block the circular iterators r and (r + _block_size) would be
123  // equal and the read...() functions wouldn't work.
124  // But anyway, who wants a delay line with no delay? Kind of useless ...
125  , _number_of_blocks(
126  std::max(size_type(2), (_max_delay + 2 * _block_size - 1) / _block_size))
127  , _data(_number_of_blocks * _block_size) // initialized with default ctor T()
128  , _data_circulator(_data.begin(), _data.end())
129  , _block_circulator(_data_circulator, _block_size)
130 {
131  assert(_block_size >= 1);
132 }

159 template<typename T, typename Container>
160 template<typename Iterator>
161 bool
162 BlockDelayLine<T, Container>::read_block(Iterator destination, size_type delay)
163  const
164 {
165  // TODO: try to get a more meaningful error message if source is not a random
166  // access iterator (e.g. when using a std::list)
167  if (!this->delay_is_valid(delay)) return false;
168  circulator source = this->get_read_circulator(delay);
169  std::copy(source, source + _block_size, destination);
170  return true;
171 }

174 template<typename T, typename Container>
175 template<typename Iterator>
176 bool
177 BlockDelayLine<T, Container>::read_block(Iterator destination
178  , size_type delay, T weight) const
179 {
180  if (!this->delay_is_valid(delay)) return false;
181  circulator source = this->get_read_circulator(delay);
182  std::transform(source, source + _block_size, destination
183  , [weight] (T in) { return in * weight; });
184  return true;
185 }
