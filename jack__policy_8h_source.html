180 template<typename X>
181 JackClient::port_t*
182 jack_policy::Xput<X>::_init_port(const parameter_map& p, jack_policy& parent)
183 {
184  auto name = std::string();
185 
186  // first, try port_name
187  if (p.has_key("port_name"))
188  {
189  name = p["port_name"];
190  }
191  else
192  {
193  // then concatenate "input_prefix"/"output_prefix" with "id"
194  // if the prefix isn't specified, it's replaced by a default string
195  // worst case: duplicate string -> port registration will fail!
196 
197  auto id = std::string();
198 
199  if (p.has_key("id"))
200  {
201  id = p.get("id", "");
202  }
203  else
204  {
205  static int next_id = 1;
206  id = str::A2S(next_id++);
207  }
208 
209  name = p.get(X::prefix_name(), X::default_prefix()) + id;
210  }
211 
212  return X::is_input
213  ? parent.register_in_port(name) : parent.register_out_port(name);
214 }
