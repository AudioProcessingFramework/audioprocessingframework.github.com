94  void create(void* (*f)(void*), void* data)
95  {
96  if (pthread_create(&_thread_id, 0, f, data))

114 template<typename F>
115 class posix_thread_policy::ScopedThread : public ThreadBase, NonCopyable
116 {
117  public:
118  ScopedThread(F f, useconds_type usleeptime)
119  : _kill_thread(false)
120  , _function(f)
121  , _usleeptime(usleeptime)
122  {
123  this->create(&_thread_aux, this);
124  }
125 
126  ~ScopedThread()
127  {
128  _kill_thread = true;
129  this->join();
130  }
131 
132  private:
133  static void* _thread_aux(void *arg)
134  {
135  static_cast<ScopedThread*>(arg)->_thread();
136  return nullptr;
137  }
138 
139  void _thread()
140  {
141  while (!_kill_thread)
142  {
143  _function();
144  usleep(_usleeptime);
145  }
146  }
147 
148  volatile bool _kill_thread;

151 };
