144  void process_commands()
145  {

147  while ((cmd = _in_fifo.pop()) != nullptr)
148  {

191 void CommandQueue::push(Command* cmd)
192 {
193  if (!_active)
194  {
195  cmd->execute();
196  _cleanup(cmd);
197  return;
198  }
199 
200  // First remove all commands from _out_fifo.
201  // This ensures that it's not going to be full which would block
202  // process_commands() and its calling realtime thread.
203  this->cleanup_commands();
204 
205  // Now push the command on _in_fifo; if the FIFO is full: retry, retry, ...
206  while (!_in_fifo.push(cmd))
207  {
208  // We don't really know if that ever happens, so we abort in debug-mode:
209  assert(false && "Error in _in_fifo.push()!");
210  // TODO: avoid this usleep()?
211  usleep(50);
212  }
213 }
