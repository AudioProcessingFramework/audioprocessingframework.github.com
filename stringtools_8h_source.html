88 {
89  bool result;
90  // first try: if input == "1" or "0":
91  input >> result;
92  if (input.fail())
93  {
94  input.clear(); // clear error flags
95  input.seekg(0); // go back to the beginning of the stream

102  if (!input.eof()) return false;

223 template<int digits, typename char_T, typename traits, typename out_T>
224 std::basic_istream<char_T, traits>&
225 convert_chars(std::basic_istream<char_T, traits>& input, out_T& output)
226 {
227  static_assert(digits > 0, "'digits' must be at least 1!");
228 
229  // if an error bit is set on the input, just return without doing anything:
230  if (input.fail()) return input;
231  // skip whitespace if std::skipws is set
232  if (input.flags() & std::ios_base::skipws) input >> std::ws;
233 
234  char_T ch[digits];
235  if (!input.read(ch, digits)) return input; // error bits are set!
236 
237  out_T factor = 1, result = 0;
238  for (int i = digits - 1; i >= 0; --i)
239  {
240  // only numbers are allowed:
241  if (ch[i] < input.widen('0') || ch[i] > input.widen('9'))
242  {
243  input.setstate(std::ios_base::failbit);
244  return input; // error bits are set!
245  }
246  // character type is implicitly cast to out_T
247  result += (ch[i] - '0') * factor;
248  factor *= 10;
249  }
250  output = result;
251  return input;
252 }

325 template<template<typename, typename, typename> class in_T,
326  typename char_T, typename traits, typename Allocator, typename out_T>
327 bool string2time(const in_T<char_T, traits, Allocator>& input, out_T& output)
328 {
329  // first of all, check if there are 0, 1 or 2 colons:
330  char colons = 0;
331  for (size_t found = 0
332  ; (found = input.find(':', found + 1)) != std::string::npos
333  ; ++colons) {}
334 
335  std::basic_istringstream<char_T> iss(input);
336  out_T seconds = 0; // initialisation is needed for the case (colons == 1)!
337 
338  if (colons == 0)
339  {
340  // no colons, but maybe suffixes like "s", "min", "h" or "ms"
341  out_T number;
342  iss >> number;
343  if (iss.fail()) return false;
344 
345  iss >> std::ws >> clear_iostate_except_eof;
346  if (iss.eof()) // that's everything, no suffixes!
347  {
348  seconds = number;
349  }
350  else // still something left ...
351  {
352  auto the_rest = std::basic_string<char_T>();
353 
354  iss >> the_rest;
355  if (iss.fail()) return false;
356 
357  iss >> std::ws >> clear_iostate_except_eof;
358  if (!iss.eof()) return false;
359 
360  // now check for possible suffixes:
361  if (the_rest == "h") seconds = number * 60 * 60;
362  else if (the_rest == "min") seconds = number * 60;
363  else if (the_rest == "s") seconds = number;
364  else if (the_rest == "ms")
365  {
366  // check if milliseconds can be represented by the type out_T:
367  // TODO: hopefully this isn't optimized away!
368  if (number / 1000 * 1000 != number) return false;
369  else seconds = number / 1000;
370  }
371  else return false; // no other suffix is allowed
372  }
373  }
374  else if (colons == 1 || colons == 2)
375  {
376  // check if there is a plus or minus sign
377  bool negative = false;
378  iss >> std::ws; // remove leading whitespace
379  if (iss.peek() == '-')
380  {
381  iss.ignore();
382  negative = true;
383  }
384  // it doesn't matter if there is a '+' sign.
385 
386  long int hours = 0;
387  int minutes; // maximum: 59
388 
389  if (colons == 1)
390  {
391  if ((iss >> minutes).fail()) return false;
392  // attention: the sign was already removed before!
393  if (minutes < 0 || minutes > 59) return false;
394  }
395  else if (colons == 2)
396  {
397  iss >> hours
398  >> std::noskipws // from now on, no whitespace is allowed
399  >> remove_colon; // read one character and check if it's a colon
400  convert_chars<2>(iss, minutes); // read minutes as two characters
401  if (iss.fail()) return false;
402  if (hours < 0) return false; // the sign was already removed before!
403  if (minutes > 59) return false;
404  }
405 
406  out_T whole_seconds;
407  out_T fraction_of_second(0);
408 
409  iss
410  >> std::noskipws // no whitespace is allowed
411  >> remove_colon; // read one character and check if it's a colon
412  convert_chars<2>(iss, whole_seconds); // read first two digits of seconds
413  if (iss.fail()) return false;
414 
415  if (whole_seconds > 59) return false;
416 
417  if (iss.peek() == '.')
418  {
419  if ((iss >> fraction_of_second).fail()) return false;
420  }
421 
422  if (!(iss >> std::ws).eof()) return false; // nothing else is allowed!
423 
424  auto the_rest = whole_seconds + fraction_of_second;
425 
426  // the seconds part must be smaller than 60
427  if (the_rest >= 60) return false;
428 
429  if (negative)
430  {
431  hours = -hours;
432  minutes = -minutes;
433  the_rest = -the_rest;
434  }
435 
436  seconds = static_cast<out_T>(hours * 60 * 60);
437  seconds += static_cast<out_T>(minutes * 60);
438  seconds += the_rest;
439  }
440  else return false; // more than three colons are a deal-breaker!
441 
442  output = seconds;
443  return true;
444 }
